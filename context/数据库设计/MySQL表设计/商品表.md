# 商品表

## DDL

```sql
-- 分类表
CREATE TABLE `saas_category` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '分类ID',
  `parent_id` BIGINT UNSIGNED NOT NULL DEFAULT 0 COMMENT '父分类ID, 0=顶级',
  `name` VARCHAR(64) NOT NULL COMMENT '分类名称',
  `level` TINYINT NOT NULL DEFAULT 1 COMMENT '层级: 1/2/3',
  `sort` INT NOT NULL DEFAULT 0 COMMENT '排序',
  `icon` VARCHAR(512) DEFAULT NULL COMMENT '图标URL',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `is_deleted` TINYINT NOT NULL DEFAULT 0,
  PRIMARY KEY (`id`),
  KEY `idx_parent_id` (`parent_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='商品分类表';

-- SPU表（标准商品单元）
CREATE TABLE `saas_spu` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT 'SPU ID',
  `spu_no` VARCHAR(32) NOT NULL COMMENT '商品编码',
  `name` VARCHAR(256) NOT NULL COMMENT '商品名称',
  `category_id` BIGINT UNSIGNED NOT NULL COMMENT '分类ID',
  `brand` VARCHAR(128) DEFAULT NULL COMMENT '品牌',
  `description` TEXT COMMENT '商品描述',
  `main_image` VARCHAR(512) DEFAULT NULL COMMENT '主图URL',
  `images` JSON DEFAULT NULL COMMENT '商品图片列表',
  `status` TINYINT NOT NULL DEFAULT 1 COMMENT '状态: 1=上架 0=下架',
  `min_price` DECIMAL(10,2) DEFAULT NULL COMMENT '最低价',
  `max_price` DECIMAL(10,2) DEFAULT NULL COMMENT '最高价',
  `sales` INT NOT NULL DEFAULT 0 COMMENT '销量',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `is_deleted` TINYINT NOT NULL DEFAULT 0,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_spu_no` (`spu_no`),
  KEY `idx_category_id` (`category_id`),
  KEY `idx_status` (`status`),
  FULLTEXT KEY `ft_name_desc` (`name`, `description`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='商品SPU表';

-- SKU表（库存量单位，规格商品）
CREATE TABLE `saas_sku` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT 'SKU ID',
  `sku_no` VARCHAR(32) NOT NULL COMMENT 'SKU编码',
  `spu_id` BIGINT UNSIGNED NOT NULL COMMENT '所属SPU ID',
  `spec_name` VARCHAR(256) DEFAULT NULL COMMENT '规格名称(如:白色/XL)',
  `spec_data` JSON DEFAULT NULL COMMENT '规格JSON(如:{"颜色":"白色","尺码":"XL"})',
  `price` DECIMAL(10,2) NOT NULL DEFAULT 0.00 COMMENT '售价',
  `cost_price` DECIMAL(10,2) DEFAULT NULL COMMENT '成本价',
  `image` VARCHAR(512) DEFAULT NULL COMMENT 'SKU图片',
  `status` TINYINT NOT NULL DEFAULT 1 COMMENT '状态: 1=正常 0=停用',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `is_deleted` TINYINT NOT NULL DEFAULT 0,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_sku_no` (`sku_no`),
  KEY `idx_spu_id` (`spu_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='商品SKU表';
```
